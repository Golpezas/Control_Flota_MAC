// 1. Componente DetailItem 
interface DetailItemProps {
    label: string;
    value: string | number | null | undefined;
}
const DetailItem: React.FC<DetailItemProps> = ({ label, value }) => (
    <div style={{ borderBottom: '1px dotted #ccc', padding: '5px 0' }}>
        <span style={{ fontWeight: 'bold', color: '#457B9D' }}>{label}:</span> {value ?? 'N/A'}
    </div>

// 2. Componente DocumentoItem
interface DocumentoItemProps {
    doc: DocumentoDigital;
    patente: string;
}
const DocumentoItem: React.FC<DocumentoItemProps> = ({ doc, patente }) => {
    const [modalIsOpen, setModalIsOpen] = useState(false);
    const [previewLoading, setPreviewLoading] = useState(false);
    const [previewError, setPreviewError] = useState<string | null>(null);

    const fileId = doc.file_id;
    const nombreArchivo = doc.nombre_archivo || 'Archivo sin nombre';

    const handlePreview = async () => {
        setPreviewLoading(true);
        setPreviewError(null);
        try {
            // L√≥gica para preview si es necesario
            setModalIsOpen(true);
        } catch (err) {
            setPreviewError('Error al cargar vista previa');
        } finally {
            setPreviewLoading(false);
        }
    };

    return (
        <div style={{ 
            display: 'flex', 
            justifyContent: 'space-between', 
            alignItems: 'center', 
            padding: '10px 0', 
            borderBottom: '1px solid #eee' 
        }}>
            <span>{doc.tipo}: {nombreArchivo}</span>
            {fileId ? (
                <div>
                    <button 
                        onClick={handlePreview} 
                        style={{ marginRight: '10px', color: '#457B9D' }}
                    >
                        Vista Previa
                    </button>
                    <a 
                        href={`${API_URL}/archivos/descargar/${fileId}`} 
                        download 
                        style={{ color: '#E63946' }}
                    >
                        Descargar
                    </a>
                </div>
            ) : (
                <span style={{ color: '#E63946' }}>Archivo no disponible</span>
            )}

            <Modal 
                isOpen={modalIsOpen} 
                onRequestClose={() => setModalIsOpen(false)} 
                style={{ 
                    content: { top: '50%', left: '50%', right: 'auto', bottom: 'auto', marginRight: '-50%', transform: 'translate(-50%, -50%)' } 
                }}
            >
                {previewLoading ? 'Cargando...' : previewError ? previewError : (
                    // L√≥gica de preview, e.g., <img> o <embed> basado en tipo
                    <div>Preview aqu√≠</div>
                )}
            </Modal>
        </div>
    );
};

// COMPONENTE PRINCIPAL: VehiculoDetail
// =================================================================
const VehiculoDetail: React.FC = () => {
    const { patente } = useParams<{ patente: string }>();
    const [vehiculo, setVehiculo] = useState<Vehiculo | null>(null);
    const [reporteCostos, setReporteCostos] = useState<ReporteCostosResponse | null>(null);
    const [gastos, setGastos] = useState<GastoUnificado[]>([]);  // Estado para gastos unificados
    const [alertas, setAlertas] = useState<Alerta[]>([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState<string | null>(null);

    // Funci√≥n para cargar todos los datos (veh√≠culo, reporte, gastos)
    const cargarDatos = useCallback(async () => {
        if (!patente) return;

        setLoading(true);
        setError(null);

        try {
            // Cargar veh√≠culo
            const vehData = await fetchVehiculoByPatente(patente);
            setVehiculo(vehData);

            // Cargar reporte de costos (ajusta fechas seg√∫n necesidad, aqu√≠ √∫ltimos 12 meses)
            const startDate = new Date();
            startDate.setMonth(startDate.getMonth() - 12);
            const endDate = new Date();
            const report = await fetchReporteVehiculo(
                patente,
                startDate.toISOString().split('T')[0],
                endDate.toISOString().split('T')[0]
            );
            setReporteCostos(report);
            setAlertas(report.alertas || []);

            // Cargar gastos unificados
            const response = await apiClient.get(`/costos/unificado/${patente}`);  // Asumir apiClient definido en vehiculos.ts
            setGastos(response.data.gastos || []);
        } catch (err) {
            setError('Error al cargar los datos del veh√≠culo.');
            console.error(err);
        } finally {
            setLoading(false);
        }
    }, [patente]);

    useEffect(() => {
        cargarDatos();
    }, [cargarDatos]);

    // Handler para borrar gasto con refresco
    const handleBorrarGasto = async (id: string, origen: "costos" | "finanzas") => {
        try {
            await borrarGastoUniversal(id, origen);
            // Refrescar datos despu√©s de borrar
            cargarDatos();
        } catch (err) {
            // El error ya se maneja en borrarGastoUniversal con alert
            console.error('Error en handleBorrarGasto:', err);
        }
    };

    if (loading) return <div>Cargando datos del veh√≠culo... ‚è≥</div>;
    if (error) return <div style={{ color: 'red' }}>‚ùå {error}</div>;
    if (!vehiculo) return <div>No se encontr√≥ el veh√≠culo.</div>;

    return (
        <div style={{ padding: '30px', maxWidth: '900px', margin: '0 auto' }}>
            <h1 style={{ borderBottom: '2px solid #ccc', paddingBottom: '10px', marginBottom: '20px', color: '#1D3557' }}>
                Detalle del Veh√≠culo: {vehiculo._id}
            </h1>

            {/* SECCI√ìN DE INFORMACI√ìN B√ÅSICA */}
            <div style={{ marginBottom: '30px' }}>
                <h2 style={{ color: '#457B9D' }}>Informaci√≥n B√°sica</h2>
                <DetailItem label="Patente Original" value={vehiculo.patente_original} />
                <DetailItem label="N¬∫ M√≥vil" value={vehiculo.nro_movil} />
                <DetailItem label="Modelo" value={vehiculo.descripcion_modelo} />
                <DetailItem label="A√±o" value={vehiculo.anio} />
                <DetailItem label="Color" value={vehiculo.color} />
                <DetailItem label="Combustible" value={vehiculo.tipo_combustible} />
                <DetailItem label="Activo" value={vehiculo.activo ? 'S√≠' : 'No'} />
            </div>

            {/* SECCI√ìN DE DOCUMENTOS DIGITALES */}
            <div style={{ marginBottom: '30px' }}>
                <h2 style={{ color: '#457B9D' }}>Documentos Digitales</h2>
                {vehiculo.documentos_digitales?.length > 0 ? (
                    vehiculo.documentos_digitales.map((doc, index) => (
                        <DocumentoItem key={index} doc={doc} patente={vehiculo._id} />
                    ))
                ) : (
                    <p>No hay documentos digitales disponibles.</p>
                )}
            </div>

            {/* SECCI√ìN DE ALERTAS */}
            <div style={{ marginBottom: '30px' }}>
                <h2 style={{ color: alertas.length > 0 ? '#E63946' : '#457B9D' }}>
                    Alertas ({alertas.length})
                </h2>
                {alertas.length > 0 ? (
                    alertas.map((alerta, index) => (
                        <div key={index} style={{ padding: '10px', background: '#fee', border: '1px solid #E63946', marginBottom: '10px' }}>
                            {alerta.mensaje}
                        </div>
                    ))
                ) : (
                    <p>No hay alertas cr√≠ticas.</p>
                )}
            </div>

            {/* SECCI√ìN DE COSTOS */}
            <div style={{ marginBottom: '30px' }}>
                <h2 style={{ color: '#457B9D' }}>Reporte de Costos</h2>
                {reporteCostos ? (
                    <>
                        <p><strong>Total General:</strong> ${reporteCostos.total_general.toLocaleString('es-AR')}</p>
                        <p><strong>Mantenimiento:</strong> ${reporteCostos.total_mantenimiento.toLocaleString('es-AR')}</p>
                        <p><strong>Infracciones:</strong> ${reporteCostos.total_infracciones.toLocaleString('es-AR')}</p>
                    </>
                ) : (
                    <p>No hay datos de costos.</p>
                )}
            </div>

            {/* SECCI√ìN DE HISTORIAL DE COSTOS - Con tabla y refresco */}
            <div style={{ 
                border: '1px solid #ddd', 
                borderRadius: '8px', 
                overflow: 'hidden', 
                backgroundColor: 'white' 
            }}>
                <h2 style={{ padding: '15px', backgroundColor: '#1D3557', color: 'white' }}>
                    Historial de Costos ({gastos.length})
                </h2>
                {gastos.length > 0 ? (
                    <table style={{ width: '100%', borderCollapse: 'collapse' }}>
                        <thead style={{ backgroundColor: '#f8f9fa' }}>
                            <tr>
                                <th style={{ padding: '12px', textAlign: 'left' }}>Fecha</th>
                                <th style={{ padding: '12px', textAlign: 'left' }}>Tipo</th>
                                <th style={{ padding: '12px', textAlign: 'left' }}>Descripci√≥n</th>
                                <th style={{ padding: '12px', textAlign: 'right' }}>Importe</th>
                                <th style={{ padding: '12px', textAlign: 'center' }}>Acciones</th>
                            </tr>
                        </thead>
                        <tbody>
                            {gastos.map((gasto, index) => (
                                <tr key={gasto.id || index} style={{ borderBottom: '1px solid #eee' }}>
                                    <td style={{ padding: '12px' }}>{gasto.fecha}</td>
                                    <td style={{ padding: '12px' }}>
                                        <span style={{ 
                                            padding: '4px 8px', 
                                            borderRadius: '12px', 
                                            backgroundColor: gasto.tipo === 'Multa' ? '#fee' : '#e6f4ea',
                                            color: gasto.tipo === 'Multa' ? '#c1121f' : '#2d6a4f',
                                            fontWeight: 'bold'
                                        }}>
                                            {gasto.tipo}
                                        </span>
                                    </td>
                                    <td style={{ padding: '12px' }}>{gasto.descripcion}</td>
                                    <td style={{ padding: '12px', textAlign: 'right', fontWeight: 'bold' }}>
                                        ${gasto.monto.toLocaleString('es-AR', { minimumFractionDigits: 2 })}
                                    </td>
                                    <td style={{ padding: '12px', textAlign: 'center' }}>
                                        <button 
                                            onClick={() => handleBorrarGasto(gasto.id, gasto.origen === "mantenimiento" ? "costos" : "finanzas")}
                                            style={{ 
                                                background: '#dc3545', 
                                                color: 'white', 
                                                border: 'none', 
                                                padding: '10px 20px', 
                                                borderRadius: '8px', 
                                                cursor: 'pointer', 
                                                fontWeight: 'bold',
                                                fontSize: '0.95em',
                                                boxShadow: '0 2px 4px rgba(0,0,0,0.2)',
                                                transition: 'background 0.3s'
                                            }}
                                            onMouseOver={(e) => e.currentTarget.style.background = '#c82333'}
                                            onMouseOut={(e) => e.currentTarget.style.background = '#dc3545'}
                                            disabled={loading}
                                        >
                                            Borrar
                                        </button>
                                    </td>
                                </tr>
                            ))}
                        </tbody>
                    </table>
                ) : (
                    <div style={{ padding: '20px', textAlign: 'center' }}>No hay costos registrados.</div>
                )}
            </div>

            {/* SECCI√ìN PARA AGREGAR NUEVO COSTO */}
            <div style={{ marginTop: '30px' }}>
                <h2 style={{ color: '#457B9D' }}>Agregar Nuevo Costo</h2>
                <CostoForm 
                    initialPatente={patente}
                    onSuccess={cargarDatos}  // Refrescar despu√©s de agregar
                />
            </div>

            <Link to="/vehiculos" style={{ display: 'block', marginTop: '30px', color: '#457B9D', textDecoration: 'none', fontWeight: 'bold' }}>
                ‚Üê Volver al Listado de Veh√≠culos
            </Link>
        </div>
    );
};

export default VehiculoDetail;
